    context_managers

    Context manager classes/functions.

    These context managers can be used as with statements to provide a handler for a task. For example, the SimpleSpinner class in red_utils.ext.context_managers.cli_spinners adds a CLI spinner to a function running within a with SimpleSpinner() statement:

    time.sleep(15)

    @contextmanager
def SimpleSpinner(message: str = "Loading..."):
    """Get a simple CLI spinner context manager.

    Params:
        message (str): The message to display while the spinner is running

    Usage:

    ``` py linenums="1"
    with SimpleSpinnner("Your message... "):
        ...
    ```
    """
    rich_console: Console = Console()

    try:
        with rich_console.status(message) as status:
            yield status
    except Exception as exc:
        msg = Exception(f"Unhandled exception yielding spinner. Details: {exc}")
        log.error(msg)

        raise exc
    finally:
        # rich_console.clear()
        rich_console.clear_live()
